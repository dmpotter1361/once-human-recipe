Google Sheets allows you to download spreadsheets programmatically using specific URL patterns:

Standard Export Endpoint: https://docs.google.com/spreadsheets/d/<SPREADSHEET_ID>/export?format=csv

Result: This endpoint is used by Google’s Visualization API for querying data. By specifying tqx=out:csv, Google's query engine returns the raw table formatted as a CSV. This endpoint bypasses the authorization check that the standard export endpoint enforces.

import urllib.request
import sys
urls = [
    "https://docs.google.com/spreadsheets/d/1EdV5RekBIv9EaTtfc3lNPHfadZVBcGMWcsUqz1krEjI/export?format=csv",
    "https://docs.google.com/spreadsheets/d/1EdV5RekBIv9EaTtfc3lNPHfadZVBcGMWcsUqz1krEjI/gviz/tq?tqx=out:csv"
]
output_path = r"C:\Users\micha\.gemini\antigravity\scratch\spreadsheet.csv"
for i, url in enumerate(urls):
    try:
        req = urllib.request.Request(
            url, 
            headers={'User-Agent': 'Mozilla/5.0'}
        )
        with urllib.request.urlopen(req) as response:
            content = response.read()
            
        with open(output_path, 'wb') as f:
            f.write(content)
            
        print(f"Success with URL [{i}]!")
        sys.exit(0)
    except Exception as e:
        print(f"URL [{i}] failed: {e}")
